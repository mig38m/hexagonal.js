Simple Usage


var canvas = document.getElementById('simple-usage'),
    map    = new Hexagonal.Map({rows: 10, cols: 14, width: canvas.width}),
    ctx    = canvas.getContext('2d');
canvas.height = map.size().height;
map.hexagons().forEach(function(hexagon) {
  var vertex = hexagon.halfEdges[0].va();
  ctx.beginPath();
  ctx.moveTo(vertex.x, vertex.y);
  hexagon.halfEdges.forEach(function(halfEdge) {
    ctx.lineTo(halfEdge.vb().x, halfEdge.vb().y);
  });
  ctx.closePath();
  ctx.stroke();
});

Wraparound


var canvas = document.getElementById('wrap-around'),
    ctx    = canvas.getContext('2d'),
    map    = new Hexagonal.Map({rows: 3, cols: 3, hexagon: {width: 48}}),
    mapCanvas    = document.createElement('canvas'),
    mapCanvasCtx = mapCanvas.getContext('2d');
map.hexagons().forEach(function(hexagon) {
  var vertex = hexagon.halfEdges[0].va();
  mapCanvasCtx.beginPath();
  mapCanvasCtx.moveTo(vertex.x, vertex.y);
  hexagon.halfEdges.forEach(function(halfEdge) {
    mapCanvasCtx.lineTo(halfEdge.vb().x, halfEdge.vb().y);
  });
  mapCanvasCtx.closePath();
  mapCanvasCtx.stroke();
});
ctx.drawImage(mapCanvas, 0, 0);
ctx.drawImage(mapCanvas, map.size().width, 0);

Offset Cursor


var canvas = document.getElementById('offset-cursor'),
    ctx    = canvas.getContext('2d'),
    map    = new Hexagonal.Map({rows: 5, cols: 5, hexagon: { radius: 24 }}),
    cursor = new Hexagonal.Cursors.OffsetCursor(map, 0, 0);
canvas.height = map.size().height;
ctx.font = "18px Verdana";
ctx.textAlign = 'center';
map.hexagons().forEach(function(hexagon) {
  var vertex = hexagon.halfEdges[0].va(),
      center = hexagon.center();
  ctx.beginPath();
  ctx.moveTo(vertex.x, vertex.y);
  hexagon.halfEdges.forEach(function(halfEdge) {
    ctx.lineTo(halfEdge.vb().x, halfEdge.vb().y);
  });
  ctx.closePath();
  ctx.stroke();
  cursor.moveTo(hexagon);
  ctx.fillText(cursor.position.x + "," + cursor.position.y, center.x, center.y + 6);
});
