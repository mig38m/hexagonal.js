Simple Usage


var canvas = document.getElementById('simple-usage'),
    map    = new Hexagonal.Map({rows: 10, cols: 14, width: canvas.width}),
    ctx    = canvas.getContext('2d');
canvas.height = map.size().height;
map.hexagons().forEach(function(hexagon) {
  var vertex = hexagon.halfEdges[0].va();
  ctx.beginPath();
  ctx.moveTo(vertex.x, vertex.y);
  hexagon.halfEdges.forEach(function(halfEdge) {
    ctx.lineTo(halfEdge.vb().x, halfEdge.vb().y);
  });
  ctx.closePath();
  ctx.stroke();
});
